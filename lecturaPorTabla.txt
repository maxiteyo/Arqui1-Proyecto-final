sec1:       LDR r5, [r0, #0]           Cargamos el valor de tabla del delay
            ADD r7, r0, #0             Guardamos el valor de la primer palabra de la secuencia
            ADD r8,r0,#33              Contador de palabras de la tabla

loop1:      LDR r9, [r7]               Lee el dato guardado en R7 (led a prender)
            STR r9, [r0, #0x800]       Saca la luz al puerto
            ADD r6,r0,r5               Guardamos en r6 el valor del delay en r5

delay:      SUBS r6, r6, #1            Restamos hasta llegar a cero
            BNE #0x18                  

            ADD r7, r7, #4             Aumentamos en 4 para pasar a la proxima palabra de tabla
            SUBS r8,r8,#1              Restamos 1 al contador de instrucciones
            ADDEQ r7,r0,#0             Si llega a cero, suma la cantidad de lineas de la tabla
            ADDEQ r8,r8,#33            Y vuelve a la primer palabra de la secuencia
            ADD R3,R0,#0x2             Contador de doble lectura

lectura:    SUBS R3,R3,#1               Lectura del puerto
            LDR R1,[R0,#0x800]         
            BNE #0x34               

            BNE #0xC                 Si no se apaga el PIN_M1 vuelve a la secuencia
